说明

本文章内容为SSM所需的配置文件

，说明了所需的jar包(可根据需求自行增加)。

SSM整合

所需jar包：

配置文件

beans配置

<?xml version="1.0" encoding="UTF-8"?>
<beans xmlns="http://www.springframework.org/schema/beans" 
    xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance" 
    xsi:schemaLocation="
        http://www.springframework.org/schema/beans 
        http://www.springframework.org/schema/beans/spring-beans.xsd">

</beans>

/resoueces目录下的applicationContext-mybatis.xml

<!-- 获取数据源 -->
<bean id="dataSource" class="org.apache.commons.dbcp2.BasicDataSource" destroy-method="close" scope="singleton">
    <property name="driverClassName" value="com.mysql.cj.jdbc.Driver"/>
    <property name="url" value="jdbc:mysql://localhost:3306/smbms?serverTimezone=UTC"/>
    <property name="username" value="root"/>
    <property name="password" value="root"/>
</bean>

配置aop

<!-- 配置aop -->
<bean id="transactionManager" class="org.springframework.jdbc.datasource.DataSourceTransactionManager">
    <property name="dataSource" ref="dataSource"/>
</bean>
<aop:aspectj-autoproxy/>
<aop:config proxy-target-class="true">
    <aop:pointcut expression="execution(* *cn.jbit.service..*(..))" id="transService"/>
    <aop:advisor advice-ref="txAdvice" pointcut-ref="transService"/>
</aop:config>
<tx:advice id="txAdvice" transaction-manager="transactionManager">
    <tx:attributes>
        <tx:method name="*" propagation="REQUIRED"/>
    </tx:attributes>
</tx:advice>

配置mybatis的SqlSessionFactoryBean

<bean id="sqlSessionFactory" class="org.mybatis.spring.SqlSessionFactoryBean">
    <property name="dataSource" ref="dataSource"/>
    <property name="configLocation" value="classpath:mybatis-config.xml"/>
</bean>

注意：如果不想配置mybatis.config.xml文件可以将configLocation更换成mapperLocations，如：

<bean id="sqlSessionFactory" class="org.mybatis.spring.SqlSessionFactoryBean">
    <property name="dataSource" ref="dataSource"/>
    <property name="mapperLocations" value="classpath:cn/jbit/dao/*.xml"/>
    <property name="typeAliasesPackage" value="cn.jbit.entity"/>
</bean>
  • mapperLocations
    • 设置dao.xml文件的路径，这样就会自动扫描映射文件
  • typeAliasesPackage
    • 设置实体类路径，在映射文件里写实体类时就不用写全路径了

配置MapperScannerConfigurer，自动注册dao接口对应的xml并配置成bean

<bean class="org.mybatis.spring.mapper.MapperScannerConfigurer">
    <property name="basePackage" value="cn.jbit.dao"></property>
</bean>

配置context标签自动扫描cn.jbit.service包下的控制器

<context:component-scan base-package="cn.jbit.service"/>

配置mybatis-config.xml

<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE configuration
  PUBLIC "-//mybatis.org//DTD Config 3.0//EN"
  "http://mybatis.org/dtd/mybatis-3-config.dtd">
<configuration>
  <settings>
    <!--开启Mybatis支持延迟加载-->
    <setting name="lazyLoadingEnabled" value="true"/>
    <!--开启Mybatis支持延迟加载-->
    <setting name="aggressiveLazyLoading" value="false"/>
    <!-- 配置log为log4j -->
    <setting name="logImpl" value="STDOUT_LOGGING"/>
  </settings>
  <typeAliases>
      <package name="cn.jbit.entity"></package>
  </typeAliases>
  <mappers>
    <package name="cn.jbit.dao"/>
  </mappers>
</configuration>

配置springmvc-servlet.xml

配置context标签自动扫描cn.jbit.controller包下的控制器

<context:component-scan base-package="cn.jbit.controller"/>

配置mvc:annotation-driven标签和消息转换器

<mvc:annotation-driven conversion-service="myConversionService">
    <mvc:message-converters register-defaults="true">
        <bean class="org.springframework.http.converter.StringHttpMessageConverter">
            <property name="supportedMediaTypes" value="text/plain;charset=UTF-8"/>
        </bean>
    </mvc:message-converters>
</mvc:annotation-driven>
<!-- 加载自定义转换器 -->
<bean id="myConversionService" class="org.springframework.context.support.ConversionServiceFactoryBean">
    <property name="converters">
        <list>
            <bean class="cn.jbit.tool.StringToDateConverter">
                <constructor-arg type="java.lang.String" value="yyyy-MM-dd"/>
            </bean>
        </list>
    </property>
</bean>

配置mvc:resources标签加载静态资源

<mvc:resources location="/statics/" mapping="/statics/**"/>

配置支持文件上传

<bean id="multipartResolver" class="org.springframework.web.multipart.commons.CommonsMultipartResolver">
    <property name="maxUploadSize" value="5000000"></property>
    <property name="defaultEncoding" value="UTF-8"></property>
</bean>

配置多视图解析器

<bean id="viewResolver" class="org.springframework.web.servlet.view.InternalResourceViewResolver">
    <property name="prefix" value="/jsp/"/>
    <property name="suffix" value=".jsp"/>
</bean>

配置拦截器

<mvc:interceptors>
    <mvc:interceptor>
        <!-- *开头的路径都拦截 -->
        <mvc:mapping path="/smbm/**"/>
        <!-- 拦截器路径 -->
        <bean class="cn.SpringMVC.interceptor.FirstInterceptor"></bean>
    </mvc:interceptor>
</mvc:interceptors>

配置web.xml

配置springmvc.xmlc扫描

<servlet>
       <servlet-name>dispatcherServlet</servlet-name>
    <servlet-class>org.springframework.web.servlet.DispatcherServlet</servlet-class>
    <init-param>
        <param-name>contextConfigLocation</param-name>
        <param-value>classpath:springmvc.xml</param-value>
    </init-param>
    <load-on-startup>1</load-on-startup>
</servlet>
<servlet-mapping>
    <servlet-name>dispatcherServlet</servlet-name>
       <url-pattern>/</url-pattern>
</servlet-mapping>

传参乱码解决

<filter>
    <filter-name>encodingFilter</filter-name>
    <filter-class>org.springframework.web.filter.CharacterEncodingFilter</filter-class>
    <init-param>
        <param-name>encoding</param-name>
        <param-value>utf-8</param-value>
    </init-param>
    <init-param>
        <param-name>forceEncoding</param-name>
        <param-value>true</param-value>
    </init-param>
</filter>
<filter-mapping>
    <filter-name>encodingFilter</filter-name>
    <url-pattern>/*</url-pattern>
</filter-mapping>

Spring监听器

<context-param>
    <param-name>contextConfigLocation</param-name>
    <param-value>classpath:applicationContext-*.xml</param-value>
</context-param>
<listener>
    <listener-class>
        org.springframework.web.context.ContextLoaderListener
    </listener-class>
</listener>

ajax返回参数的方式更换

在ajax返回参数时，可以试着返回一个自定义的类。类里面有执行状态(int)，状态说明(String)，需要传递的对象(Object)这三个属性。在jsp页面就可以根据执行状态来觉得执行什么内容，

比如说传递了自定义的类，根据类的有参构造器，传入了执行状态：200，状态说明：执行成功，传递的对象：User。这时候在jsp页面就可以判断，如果data.执行状态=200，就执行登录操作

首先创建一个枚举类

enum ResultCode

然后在枚举类里创建几个数据的状态，例如执行成功(SUCCESS(200,”成功”))，执行失败(FAIL(303,”失败”))，对象为空(EMPTY(250,”空值”))。这个数据状态是可以根据需求添加的

SUCCESS(200,"成功"),
FAIL(303,"失败"),
EMPTY(250,"空值"),
ERROR(101,"错误");

在枚举类里创建两个属性：执行状态和状态说明。并提供set和get方法

private int code ;
private String desc;

最后提供有参和无参的构造器

创建一个自定义状态类：ResponseMsg

并让该类实现Serializable。

在该类里面有三个私有属性：

Integer status
String msg
Object data

并提供对应的set和get方法

根据需求提供对应的有参和无参构造器

根据枚举类里的数据状态写对应的静态方法

public static  ResponseMsg BySuccess(String msg,Object data) {
    return new ResponseMsg(ResultCode.SUCCESS.getCode(),msg,data);
}

这个方法就是执行成功需要返回的自定义类方法。在返回时，传入说明和一个对象。同时，在调用这个方法时会自动获取对应的枚举：ResultCode.SUCCESS.getCode()

public static ResponseMsg ByFail(String msg) {
    return new ResponseMsg(ResultCode.FAIL.getCode(),msg);
}

这个方法就是执行失败返回的自定义方法，在返回时，只需要提供说明即可。这样返回的就是对应的枚举和一个说明

注意：写不同的方法时，需要改变对应的枚举，比如执行成功就是传入的ResultCode.SUCCESS.getCode()，执行失败就是传入的ResultCode.FAIL.getCode()。

在写不同的方法时需要不同的有参构造器，比如说执行成功就需要全参构造器
