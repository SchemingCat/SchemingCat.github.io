使用SpringSecurity

依赖

首先导入所需依赖

<dependency>
    <groupId>org.springframework.boot</groupId>

</dependency>
<dependency>
    <groupId>org.springframework.boot</groupId>

指定登录账号

添加security后项目就会受到保护，任何操作(访问控制层)都需要登录才能执行。默认会存在一个账号，用户名为user，密码在控制台自动生成

登录账号一般从数据库获取，这里一般用于指定测试的账号

在配置文件中指定账号

spring.security.user.name=testUser
spring.security.user.password=123456
spring.security.user.roles=admin

在Java中指定账号

首先让类继承WebSecurityConfigurerAdapter，然后重写configure(AuthenticationManagerBuilder auth)方法

auth
.inMemoryAuthentication()
.withUser("admin") // 用户名
.password("123")   // 密码
.roles("ADMIN")    // 角色
.and()             // 指定多个账号
.withUser("user")        
.password("123")
.roles("ADMIN", "USER") // 指定多个角色

指定登录密码是否加密

@Bean
public PasswordEncoder passwordEncoder(){
    return NoOpPasswordEncoder.getInstance();
}

绑定权限所能执行的操作

首先在方法上加入区别，如将/hello改为/admin/hello。然后去重写WebSecurityConfigurerAdapter的方法，将访问地址和权限进行绑定

继承WebSecurityConfigurerAdapter后重写cofigure(HttpSecurity http)方法。使用http.authorizeRequests()完成授权

http.authorizeRequests()
.antMathchers("/admin/**")  
.hasRole("ADMIN")                       // 表示只有拥有ADMIN权限的用户才能访问/admin开头控制层
.antMathchers("/user/**")
.access("hasAnyRole('ADMIN','USER')")   // 表示/user开头的控制层既可以由ADMIN访问，也能用USER访问
.anyRequest()           // 表示任何操作都需要进行验证。
.authenticated()
.and()
.formLogin()
.usernameParameter("userName")          // 指定验证时的传递的用户名名称
.passwordParameter("userPassword")      // 指定验证时的传递的密码名称
.loginPage("/login")                    // 指定登录页面的控制层地址
.loginProcessingUrl("/user/login")      // 指定处理登录的控制层地址
.permitAll()
.and()
.logout()
.logoutUrl()
.clearAuthentication(true)
.invalidateHttpSession(true)            // 禁用Session
.permitAll()                            // 登录相关的接口不用认证
.and()
.csrf()
.disable();

密码加密

首先指定密码类型

@Bean
public PasswordEncoder passwordEncoder(){

方法安全

首先在类上面加上

@EnableGlobalMethodSecurity(prePostEnabled = true, securedEnabled = true)

prePostEnabled = true会解锁@PreAuthorize和@PostAuthorize两个注解。@PreAuthorize在方法执行前进行验证，@PostAuthorize在方法执行后进行验证

securedEnabled = true会解锁@Secured注解。

使用：

public class MethodService{
    @Secured("ROLE_ADMIN")
    public String num1(){
        return "";
    }

    @PreAuthorize("hasRole('ADMIN') and hasRole("USER")")
    public String num2(){
        return "";
    }

    @PreAuthorize("hasRole('ADMIN','USER')")
    public String num3(){
        return "";
    }
}
  • @Secured(“ROLE_ADMIN”)
    • 表示访问该方法必须有ADMIN角色(注意：角色前面需要加上ROLE_)
  • @PreAuthorize(“hasRole(‘ADMIN’) and hasRole(“USER”)”)
    • 表示访问该方法必须有ADMIN和USER角色
  • @PreAuthorize(“hasRole(‘ADMIN’,’USER’)”)
    • 表示访问该方法必须是’ADMIN’和’USER’其中一个角色

首先让实体类继承UserDatails

// 使用lombok
@Date
public class User implements UserDateils{
    private Integer id;
    private String username;
    private String password;
    private Boolean locked;
    private Boolean enabled;

    private List<Role> roleLise = new ArrayList<>)();

    // 重写UserDateils的方法
    getUserName();  // 获取当前对象的用户名
    getPassword();  // 获取当前对象的密码
    getAuthorities();   // 获取当前对象的角色信息
    isAccoutNonExpired();   // 当前账号是否过期
    isAccoutNonLocked();    // 当前账号是否锁定
    isCredentialNonExpired();   // 当前账号密码是否过期
    isEnabled();    // 当前账号是否可用

    @Override
    public Collectio<? extends GrantedAuthority> getAuthorities(){
        List<SimpleGrantedAuthority> authorities = new ArrayList<>();
        // 遍历查询登录后出来的Role集合，获取登录用户的所有Role权限名称
        for(Role role : roleLise){
            authorities.add(new SimpleGrantedAuthority(role.getName()))
        }
        return authorities;
    }
}

写一个service，实现UserDatailsService接口

@Service
public class UserService implements UserDatailsService{
    @Resource
    private UserMapper userMapper;

    @Override
    public UserDetails loadUserByUsername(String name) throws UsernameNotFoundException{
        User user = userMapper.getUserByName(name);
        if(user == null){
            throws new UsernameNotFoundException("用户不存在");
        }
        // 获取登录后所拥有的权限。在实体类中进行遍历，将权限名称存入SimpleGrantedAuthority
        user.setRoles(userMapper.getUserRoleById(user.getId()));
        return user;
    }

}

在基础WebSecurityConfigurerAdapter的类中进行配置

@Configuration
public class MyWebSecurityConfig extends WebSecurityConfigurerAdapter{
    @Resource
    private UserService userService;

    @Bean
    public PasswordEncoder passwordEncoder(){
        return new BCryptPasswordEncoder(5);
    }

    @Override
    protected void configure(AuthenticationManagerBuilder auth){
        auth.userDetailService(userService);
    }

    @Override
    protected void configure(HttpSecurity http){
        http.authorizeRequests()
            .antMathchers("/admin/**").hasRole("ADMIN")                       // 表示只有拥有ADMIN权限的用户才能访问/admin开头控制层
            .antMathchers("/user/**").access("hasAnyRole('ADMIN','USER')")   // 表示/user开头的控制层既可以由ADMIN访问，也能用USER访问
            .anyRequest()           // 表示任何操作都需要进行验证。
            .authenticated()
            .and()
            .formLogin()
            .usernameParameter("userName")          // 指定验证时的传递的用户名名称
            .passwordParameter("userPassword")      // 指定验证时的传递的密码名称
            .loginPage("/login")                    // 指定登录页面的控制层地址
            .loginProcessingUrl("/user/login")      // 指定处理登录的控制层地址
            .permitAll()
            .and()
            .logout()
            .logoutUrl()
            .clearAuthentication(true)
            .invalidateHttpSession(true)            // 禁用Session
            .permitAll()                            // 登录相关的接口不用认证
            .and()
            .csrf()
            .disable();
    }
}

从数据库获取权限及权限对应的操作

从数据库获取权限访问信息(configure(HttpSecurity http)配置信息)

创建CustomFilterInvocationSecurityMetadataSource获取当前请求URL所需权限

public class CustomFilterInvocationSecurityMetadataSource implements FilterInvocationSecurityMetadataSource {
    @Resource
    private MenuMapper menuMapper;

    private AntPathMatcher antPathMatcher = new AntPathMatcher();

    @Override
    public Collection<ConfigAttribute> getAttributes(Object obj) throws IllegalArgumentException {
        // 获取请求地址
        String requestUrl = ((FilterInvocation)obj).getRequestUrl();
        List<Menu> menuList = menuMapper.getAllMenu();
        for(Menu menu : menuList){
            if(antPathMatcher.match(menu.getPattern(), requestUrl)){
                List<Role> roleList = menu.getRoleList();
                String[] roleString = new String[roleList.size()];
                for(int i=0;i<roleString.length;i++){
                    roleString[i] = roleList.get(i).getName();
                }
                return SecurityConfig.createList(roleString);
            }
        }
        return SecurityConfig.createList("ROLE_LOGIN");
    }

    @Override
    public Collection<ConfigAttribute> getAllConfigAttributes() {
        return null;
    }

    @Override
    public boolean supports(Class<?> aClass) {
        return FilterInvocation.class.isAssignableFrom(aClass);
    }
}

创建AccessDecisionManager，获取当前登录用户的所有权限

public class AccessDecisionManager implements org.springframework.security.access.AccessDecisionManager {
    @Override
    public void decide(Authentication authentication, Object o, Collection<ConfigAttribute> collection) throws AccessDeniedException, InsufficientAuthenticationException {
        Collection<? extends GrantedAuthority> authorities = authentication.getAuthorities();
        for(ConfigAttribute configAttribute : collection){
            if("ROLE_LOGIN".equals(configAttribute.getAttribute()) && authentication instanceof UsernamePasswordAuthenticationToken){
                return;
            }
            for(GrantedAuthority authority : authorities){
                if(configAttribute.getAttribute().equals(authority.getAuthority())){
                    return;
                }
            }
        }
        throw new AccessDeniedException("权限不足");
    }

    @Override
    public boolean supports(ConfigAttribute configAttribute) {
        return true;
    }

    @Override
    public boolean supports(Class<?> aClass) {
        return true;
    }
}
